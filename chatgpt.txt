const playMusic = (track, autoPlay = false) => {
    const encodedTrack = encodeURIComponent(track);
    const url = "/songs/" + encodedTrack;
    currentsong.src = "/songs/" + encodedTrack;
    console.log("Attempting to load:", url);

    // Select the songinfo element by class
    const songinfoElement = document.querySelector(".songinfo");

    if (!autoPlay) {
        songinfoElement.innerHTML = decodeURI(track);
        document.querySelector(".songtime").innerHTML = "0:00/0:00";
        
    }

    if (autoPlay) {
        currentsong.play().catch(error => {
            console.error("Failed to load audio:", error);
        });
    }
};
